def plotprobs(Ps, allnum, nums, previous, **kwargs):
    probs = nums/allnum
    sigma = sqrt(allnum * probs * (1-probs))/allnum
    
    if previous is not None:
        data = probs+previous
    else:
        data = probs
        
    plt.plot(Ps, data, **kwargs)
    return data
