def extended(d):
    l = log10(d)
    dl = l[-1] - l[-2]
    l = l - 0.5*dl
    l = np.concatenate([l, [l[-1]+dl]])
    return 10**l

Ns = [16, 32, 64, 128, 256, 512, 1024, 2048]
